search_terms = list(dataset["search_term"]) 
def aggregate(search_terms): 
    freq = {} 
    for term in search_terms: 
        if term not in freq:
            freq[term] = 1 
        else: 
            freq[term] += 1 
    return freq

def list_stats(array): 
    minv = min(array)
    maxv = max(array)
    sumv = sum(array)
    meanv = sumv / len(array)
    stdev = statistics.stdev(array)
    variance = statistics.variance(array)
    cv = stdev / meanv
    return {
        "min" : minv, 
        "max" : maxv, 
        "sum" : sumv, 
        "mean" : meanv, 
        "stdev" : stdev, 
        "variance" : variance, 
        "cv" : cv
    }
